  * PrimaryName: nvarchar
  * YHSIId : nvarchar

  * ShowInRegister: bit

  Records: nvarchar

}

  <only string used>
  4 : Federal

}

  1 : Building
  2 : District

}

  <only string used>

  2: WHBR

  --
  * PlaceId : int

  1 : Political
